# CRM Data Pipeline — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Extract TAX Invoice + Quotation xlsx files from Google Drive into PostgreSQL, sync summaries to Google Sheets, expose a webhook for real-time triggers.

**Architecture:** Next.js 15 API route `/api/gdrive` receives Google Drive Watch webhook, spawns a Python worker via `child_process.spawn`. The Python worker downloads the file, parses filename + xlsx content, upserts customers/salespersons/products into PostgreSQL, and appends a summary row to Google Sheets. A separate `batch_import.py` script handles historical data by listing all files in the Drive folders and passing each through the same pipeline.

**Tech Stack:** Python 3.11+ (openpyxl, psycopg2-binary, google-api-python-client, pytest), Next.js 15 (App Router, TypeScript), PostgreSQL (DigitalOcean), Prisma (schema + migrations only — DB access from Python uses psycopg2 directly, Next.js CRM uses Prisma Client), Google Drive API v3, Google Sheets API v4.

```python
# extraction/parsers/filename_parser.py
import re
from datetime import date
from dataclasses import dataclass
from typing import Optional


@dataclass
class FilenameMetadata:
    doc_type: str
    doc_number: str
    doc_date: date
    channel: str
    salesperson: str
    payment_status: str          # 'paid' | 'pending'
    ref_doc_number: Optional[str]
    customer_short: str
    province: str


_TI_RE = re.compile(
    r"TI_B No (?P<doc_number>\S+) (?P<date>\d{2}-\d{2}-\d{4}) "
    r"(?P<channel>\S+) (?P<salesperson>[^(\-]+?)\s*"
    r"\((?P<payment>[^)]*)\)\((?P<ref>[^)]*)\)\s*"
    r"(?P<customer>.+?)\s+(?P<province>\S+)$",
    re.UNICODE,
)

_QT_RE = re.compile(
    r"Quotation No (?P<doc_number>\S+) (?P<date>\d{2}-\d{2}-\d{4}) "
    r"(?P<channel>\S+) (?P<salesperson>[^(]+?)\s*"
    r"\((?P<payment>[^)]*)\)\s*"
    r"(?P<customer>.+?)\s+(?P<province>\S+)$",
    re.UNICODE,
)


def _parse_date(raw: str) -> date:
    day, month, year = raw.split("-")
    return date(int(year), int(month), int(day))


def parse_filename(filename: str) -> FilenameMetadata:
    stem = filename.rsplit(".", 1)[0]

    if stem.startswith("TI_B"):
        m = _TI_RE.match(stem)
        if not m:
            raise ValueError(f"Cannot parse TAX Invoice filename: {filename!r}")
        doc_type = "tax_invoice"
        ref_raw = m.group("ref").strip()
        ref_doc_number = None if ref_raw == "--" else ref_raw
    elif stem.startswith("Quotation"):
        m = _QT_RE.match(stem)
        if not m:
            raise ValueError(f"Cannot parse Quotation filename: {filename!r}")
        doc_type = "quotation"
        ref_doc_number = None
    else:
        raise ValueError(f"Cannot parse filename (unknown type): {filename!r}")

    payment_raw = m.group("payment").strip()
    payment_status = "paid" if "-PAID-" in payment_raw else "pending"

    return FilenameMetadata(
        doc_type=doc_type,
        doc_number=m.group("doc_number").strip(),
        doc_date=_parse_date(m.group("date")),
        channel=m.group("channel").strip(),
        salesperson=m.group("salesperson").strip(),
        payment_status=payment_status,
        ref_doc_number=ref_doc_number,
        customer_short=m.group("customer").strip(),
        province=m.group("province").strip(),
    )
```

```python
# extraction/parsers/xlsx_parser.py
import openpyxl
from dataclasses import dataclass, field
from decimal import Decimal, InvalidOperation
from typing import Optional


@dataclass
class CustomerData:
    name: str
    tax_id: Optional[str]
    address: str


@dataclass
class ItemData:
    line_no: int
    description: str
    quantity: Decimal
    unit: str
    unit_price: Decimal
    total: Decimal


@dataclass
class DocumentData:
    doc_number: str
    doc_date_raw: object          # datetime or string from cell
    customer: CustomerData
    items: list
    subtotal: Decimal
    vat: Decimal
    total: Decimal
    notes: str


def _cell_str(ws, row: int, col: int) -> str:
    v = ws.cell(row=row, column=col).value
    return str(v).strip() if v is not None else ""


def _cell_decimal(ws, row: int, col: int) -> Decimal:
    v = ws.cell(row=row, column=col).value
    if v is None:
        return Decimal("0")
    try:
        return Decimal(str(v))
    except InvalidOperation:
        return Decimal("0")


def _split_customer(raw: str):
    """'บริษัท เคไอที ดีไซน์ (สำนักงานใหญ่) TAX ID : 0105564167642' → (name, tax_id)"""
    for sep in ("TAX ID :", "TAX ID:", "เลขที่ผู้เสียภาษี"):
        if sep in raw:
            parts = raw.split(sep, 1)
            name = parts[0].strip().rstrip("()").strip()
            # tax_id is the first contiguous digit string after the separator
            import re
            m = re.search(r"\d{13}", parts[1])
            tax_id = m.group(0) if m else parts[1].strip().split()[0]
            return name, tax_id
    return raw.strip(), None


def parse_tax_invoice(filepath: str) -> DocumentData:
    wb = openpyxl.load_workbook(filepath, data_only=True)
    ws = wb.worksheets[0]

    doc_number = _cell_str(ws, 9, 15)      # O9
    doc_date_raw = ws.cell(row=9, column=25).value  # Y9

    customer_raw = _cell_str(ws, 11, 2)    # B11
    name, tax_id = _split_customer(customer_raw)
    addr_parts = [_cell_str(ws, 12, 2), _cell_str(ws, 13, 2)]
    address = " ".join(p for p in addr_parts if p)

    items = []
    for row in range(18, 46, 2):           # rows 18,20,22,...,44
        desc = _cell_str(ws, row, 2)       # B
        if not desc:
            continue
        qty = _cell_decimal(ws, row, 17)   # Q
        unit = _cell_str(ws, row, 20)      # T
        price = _cell_decimal(ws, row, 23) # W
        total_cell = _cell_decimal(ws, row, 26)  # Z
        # prefer cell value; fallback to qty*price
        total = total_cell if total_cell else (qty * price).quantize(Decimal("0.01"))
        items.append(ItemData(
            line_no=len(items) + 1,
            description=desc,
            quantity=qty,
            unit=unit,
            unit_price=price,
            total=total,
        ))

    # Grand total at X49 (col 24, row 49) — VAT-inclusive
    total_incl = _cell_decimal(ws, 49, 24)
    if total_incl == 0 and items:
        # Fallback: sum items (cached formula value missing)
        total_incl = sum(i.total for i in items)

    subtotal = (total_incl / Decimal("107") * Decimal("100")).quantize(Decimal("0.01"))
    vat = (total_incl / Decimal("107") * Decimal("7")).quantize(Decimal("0.01"))

    notes = _cell_str(ws, 51, 3)           # C51

    return DocumentData(
        doc_number=doc_number,
        doc_date_raw=doc_date_raw,
        customer=CustomerData(name=name, tax_id=tax_id, address=address),
        items=items,
        subtotal=subtotal,
        vat=vat,
        total=total_incl,
        notes=notes,
    )


def parse_quotation(filepath: str) -> DocumentData:
    # Quotation uses same sheet structure as TAX Invoice.
    # Cell map confirmed during implementation — update if different.
    return parse_tax_invoice(filepath)
```

```python
# extraction/db.py
import psycopg2
from typing import Optional
from parsers.xlsx_parser import CustomerData


def upsert_salesperson(conn, name: str, channel: str) -> int:
    with conn.cursor() as cur:
        cur.execute("""
            INSERT INTO salespersons (name, channel)
            VALUES (%s, %s)
            ON CONFLICT DO NOTHING
            RETURNING id
        """, (name, channel))
        row = cur.fetchone()
        if row:
            return row[0]
        cur.execute("SELECT id FROM salespersons WHERE name = %s", (name,))
        return cur.fetchone()[0]


def upsert_customer(conn, customer: CustomerData, province: str) -> int:
    with conn.cursor() as cur:
        if customer.tax_id:
            cur.execute("""
                INSERT INTO customers (name, tax_id, address, province)
                VALUES (%s, %s, %s, %s)
                ON CONFLICT (tax_id) DO UPDATE
                  SET name = EXCLUDED.name,
                      address = EXCLUDED.address,
                      updated_at = NOW()
                RETURNING id
            """, (customer.name, customer.tax_id, customer.address, province))
        else:
            # No tax_id: match by exact name
            cur.execute("""
                INSERT INTO customers (name, address, province)
                VALUES (%s, %s, %s)
                ON CONFLICT DO NOTHING
                RETURNING id
            """, (customer.name, customer.address, province))
            row = cur.fetchone()
            if not row:
                cur.execute("SELECT id FROM customers WHERE name = %s LIMIT 1", (customer.name,))
                row = cur.fetchone()
            return row[0]
        return cur.fetchone()[0]


def insert_document(conn, d: dict) -> int:
    with conn.cursor() as cur:
        cur.execute("""
            INSERT INTO documents (
                doc_type, doc_number, doc_date, channel, salesperson_id,
                payment_status, ref_doc_number, customer_id,
                subtotal, vat, total, notes, gdrive_file_id, gdrive_filename
            ) VALUES (
                %(doc_type)s, %(doc_number)s, %(doc_date)s, %(channel)s, %(salesperson_id)s,
                %(payment_status)s, %(ref_doc_number)s, %(customer_id)s,
                %(subtotal)s, %(vat)s, %(total)s, %(notes)s, %(gdrive_file_id)s, %(gdrive_filename)s
            )
            ON CONFLICT (gdrive_file_id) DO NOTHING
            RETURNING id
        """, d)
        row = cur.fetchone()
        if not row:
            # Already exists — return existing id
            cur.execute("SELECT id FROM documents WHERE gdrive_file_id = %s", (d["gdrive_file_id"],))
            row = cur.fetchone()
        return row[0]


def insert_document_items(conn, document_id: int, items: list, product_ids: list) -> None:
    with conn.cursor() as cur:
        for item, product_id in zip(items, product_ids):
            cur.execute("""
                INSERT INTO document_items (
                    document_id, line_no, description, quantity, unit, unit_price, total, product_id
                ) VALUES (%s, %s, %s, %s, %s, %s, %s, %s)
            """, (
                document_id, item.line_no, item.description,
                item.quantity, item.unit, item.unit_price, item.total, product_id,
            ))


def log_sync(conn, gdrive_file_id: str, filename: str, status: str, error_msg: Optional[str]) -> None:
    with conn.cursor() as cur:
        cur.execute("""
            INSERT INTO sync_log (gdrive_file_id, filename, status, error_msg)
            VALUES (%s, %s, %s, %s)
        """, (gdrive_file_id, filename, status, error_msg))
    conn.commit()


def is_already_synced(conn, gdrive_file_id: str) -> bool:
    with conn.cursor() as cur:
        cur.execute("""
            SELECT 1 FROM sync_log
            WHERE gdrive_file_id = %s AND status = 'success'
            LIMIT 1
        """, (gdrive_file_id,))
        return cur.fetchone() is not None
```

```python
# extraction/product_matcher.py
import re
from typing import Optional


def match_product_id(conn, description: str) -> Optional[int]:
    """Match description against product_transform_rules. Returns product_id or None."""
    with conn.cursor() as cur:
        cur.execute("""
            SELECT pattern, product_id, priority
            FROM product_transform_rules
            ORDER BY priority DESC
        """)
        rules = cur.fetchall()

    for pattern, product_id, _ in rules:
        try:
            if re.search(pattern, description, re.UNICODE | re.IGNORECASE):
                return product_id
        except re.error:
            continue   # skip malformed patterns

    return None
```

```python
import io
import json
import os
from typing import Generator
from google.oauth2 import service_account
from googleapiclient.discovery import build
from googleapiclient.http import MediaIoBaseDownload

_SCOPES = [
    "https://www.googleapis.com/auth/drive.readonly",
]


def _get_service():
    sa_json = os.environ["GOOGLE_SERVICE_ACCOUNT_JSON"]
    info = json.loads(sa_json)
    creds = service_account.Credentials.from_service_account_info(info, scopes=_SCOPES)
    return build("drive", "v3", credentials=creds)


def download_file(file_id: str, dest_path: str) -> None:
    """Download a Drive file by ID to dest_path."""
    service = _get_service()
    request = service.files().get_media(fileId=file_id)
    with open(dest_path, "wb") as fh:
        downloader = MediaIoBaseDownload(fh, request)
        done = False
        while not done:
            _, done = downloader.next_chunk()


def list_files_in_folder(folder_id: str, mime_type: str = None) -> Generator[dict, None, None]:
    """Yield file dicts {id, name} for all files in folder (recursive)."""
    service = _get_service()
    query = f"'{folder_id}' in parents and trashed = false"
    if mime_type:
        query += f" and mimeType = '{mime_type}'"

    page_token = None
    while True:
        resp = service.files().list(
            q=query,
            fields="nextPageToken, files(id, name, mimeType)",
            pageToken=page_token,
            pageSize=100,
        ).execute()
        for f in resp.get("files", []):
            if f["mimeType"] == "application/vnd.google-apps.folder":
                yield from list_files_in_folder(f["id"], mime_type)
            else:
                yield f
        page_token = resp.get("nextPageToken")
        if not page_token:
            break
```

```python
# extraction/extract_file.py
"""
Usage: python extract_file.py --file-id DRIVE_FILE_ID --filename ORIGINAL_FILENAME

Downloads the file, parses it, stores in PostgreSQL, appends to Google Sheets.
Exits 0 on success, 1 on error. All errors are logged to sync_log — no crashes.
"""
import argparse
import os
import sys
import tempfile
import traceback
from datetime import date

import psycopg2
from dotenv import load_dotenv

load_dotenv()

from db import (
    is_already_synced, upsert_salesperson, upsert_customer,
    insert_document, insert_document_items, log_sync,
)
from gdrive_client import download_file
from parsers.filename_parser import parse_filename
from parsers.xlsx_parser import parse_tax_invoice, parse_quotation
from product_matcher import match_product_id
from sheets_client import append_document_row


def _thai_date_to_gregorian(raw) -> date:
    """Convert Y9 cell value (Thai Buddhist Era) to Gregorian date."""
    if isinstance(raw, date):
        # openpyxl sometimes returns a datetime object
        # Thai BE year in xlsx = Gregorian + 543, but openpyxl parses it as-is
        # Check if year > 2500 → subtract 543
        if raw.year > 2500:
            return raw.replace(year=raw.year - 543)
        return raw
    # String like "15/05/2569"
    if isinstance(raw, str) and "/" in raw:
        d, m, y = raw.split("/")
        year = int(y)
        if year > 2500:
            year -= 543
        return date(year, int(m), int(d))
    return None


def process(file_id: str, filename: str) -> None:
    db_url = os.environ["DATABASE_URL"]
    sheet_id = os.environ["GOOGLE_SHEETS_ID"]

    conn = psycopg2.connect(db_url)
    conn.autocommit = False

    try:
        if is_already_synced(conn, file_id):
            print(f"[skip] already synced: {filename}")
            conn.close()
            return

        # Download to temp file
        with tempfile.NamedTemporaryFile(suffix=".xlsx", delete=False) as tmp:
            tmp_path = tmp.name
        download_file(file_id, tmp_path)

        # Parse filename
        meta = parse_filename(filename)

        # Parse xlsx
        if meta.doc_type == "tax_invoice":
            doc_data = parse_tax_invoice(tmp_path)
        else:
            doc_data = parse_quotation(tmp_path)

        # Convert Thai date
        doc_date = _thai_date_to_gregorian(doc_data.doc_date_raw) or meta.doc_date

        # Upsert salesperson
        sp_id = upsert_salesperson(conn, meta.salesperson, meta.channel)

        # Upsert customer
        cust_id = upsert_customer(conn, doc_data.customer, meta.province)

        # Match products
        product_ids = [match_product_id(conn, item.description) for item in doc_data.items]

        # Insert document
        doc_id = insert_document(conn, {
            "doc_type": meta.doc_type,
            "doc_number": meta.doc_number,
            "doc_date": doc_date,
            "channel": meta.channel,
            "salesperson_id": sp_id,
            "payment_status": meta.payment_status,
            "ref_doc_number": meta.ref_doc_number,
            "customer_id": cust_id,
            "subtotal": float(doc_data.subtotal),
            "vat": float(doc_data.vat),
            "total": float(doc_data.total),
            "notes": doc_data.notes,
            "gdrive_file_id": file_id,
            "gdrive_filename": filename,
        })

        insert_document_items(conn, doc_id, doc_data.items, product_ids)
        conn.commit()

        # Append to Google Sheets
        append_document_row(sheet_id, {
            "doc_type": meta.doc_type,
            "doc_number": meta.doc_number,
            "doc_date": doc_date,
            "channel": meta.channel,
            "salesperson": meta.salesperson,
            "customer_name": doc_data.customer.name,
            "total": float(doc_data.total),
            "payment_status": meta.payment_status,
            "gdrive_filename": filename,
        })

        log_sync(conn, file_id, filename, "success", None)
        print(f"[ok] {filename}")

    except Exception as e:
        conn.rollback()
        error_msg = traceback.format_exc()
        log_sync(conn, file_id, filename, "error", str(e))
        print(f"[error] {filename}: {e}", file=sys.stderr)
        conn.close()
        sys.exit(1)

    finally:
        conn.close()
        try:
            os.unlink(tmp_path)
        except Exception:
            pass


if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--file-id", required=True)
    parser.add_argument("--filename", required=True)
    args = parser.parse_args()
    process(args.file_id, args.filename)
```

```python
# extraction/batch_import.py
"""
Usage:
  python batch_import.py --folder tax_invoices
  python batch_import.py --folder quotations
  python batch_import.py --all
  python batch_import.py --folder tax_invoices --year 2024
"""
import argparse
import csv
import json
import os
import time
from datetime import datetime

import psycopg2
from dotenv import load_dotenv

load_dotenv()

from db import is_already_synced
from extract_file import process
from gdrive_client import list_files_in_folder


def run_batch(folder_key: str, year_filter: int = None):
    folder_ids = json.loads(os.environ["GOOGLE_DRIVE_FOLDER_IDS"])
    folder_id = folder_ids[folder_key]

    db_url = os.environ["DATABASE_URL"]
    conn = psycopg2.connect(db_url)

    files = list(list_files_in_folder(folder_id))
    total = len(files)
    print(f"Found {total} files in {folder_key}")

    counts = {"success": 0, "skip": 0, "error": 0}
    errors = []

    for i, f in enumerate(files, 1):
        file_id = f["id"]
        filename = f["name"]

        # Year filter: check filename contains the year
        if year_filter and str(year_filter) not in filename:
            counts["skip"] += 1
            continue

        if is_already_synced(conn, file_id):
            counts["skip"] += 1
            print(f"[{i}/{total}] skip {filename}")
            continue

        print(f"[{i}/{total}] processing {filename}")
        try:
            process(file_id, filename)
            counts["success"] += 1
        except SystemExit:
            counts["error"] += 1
            errors.append({"file_id": file_id, "filename": filename})
        except Exception as e:
            counts["error"] += 1
            errors.append({"file_id": file_id, "filename": filename, "error": str(e)})

        time.sleep(0.1)   # Google Drive API rate limit

    conn.close()

    print(f"\nDone: success={counts['success']} skip={counts['skip']} error={counts['error']}")

    if errors:
        report_path = f"error_report_{folder_key}_{datetime.now().strftime('%Y%m%d_%H%M%S')}.csv"
        with open(report_path, "w", newline="", encoding="utf-8") as f:
            w = csv.DictWriter(f, fieldnames=["file_id", "filename", "error"])
            w.writeheader()
            w.writerows(errors)
        print(f"Error report: {report_path}")


if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--folder", choices=["tax_invoices", "quotations"])
    parser.add_argument("--all", action="store_true")
    parser.add_argument("--year", type=int)
    args = parser.parse_args()

    if args.all:
        run_batch("tax_invoices", args.year)
        run_batch("quotations", args.year)
    elif args.folder:
        run_batch(args.folder, args.year)
    else:
        parser.error("Provide --folder or --all")
```

```typescript
// src/app/api/gdrive/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { spawn } from 'child_process'
import path from 'path'

export async function POST(request: NextRequest): Promise<NextResponse> {
  // Google Drive sends state='sync' on initial setup — acknowledge and return
  const state = request.headers.get('x-goog-resource-state')
  if (state === 'sync') {
    return new NextResponse(null, { status: 200 })
  }

  // Validate our secret token
  const token = request.headers.get('x-goog-channel-token')
  if (token !== process.env.GDRIVE_WEBHOOK_TOKEN) {
    return new NextResponse('Unauthorized', { status: 401 })
  }

  const fileId = request.headers.get('x-goog-resource-id')
  const filename = request.headers.get('x-goog-resource-uri') ?? ''

  if (!fileId || state !== 'update') {
    return new NextResponse(null, { status: 200 })
  }

  // Extract filename from URI or use file ID as fallback
  // Real filename will be fetched by Python via Drive API
  const pythonExe = process.env.PYTHON_VENV_PATH ?? 'python'
  const extractDir = process.env.EXTRACTION_DIR ?? './extraction'
  const scriptPath = path.join(extractDir, 'extract_file.py')

  // Spawn async — do not await (webhook must return fast)
  const child = spawn(pythonExe, [
    scriptPath,
    '--file-id', fileId,
    '--filename', filename,
  ], {
    cwd: extractDir,
    detached: true,
    stdio: 'ignore',
  })
  child.unref()

  return new NextResponse(null, { status: 200 })
}
```

> **Note:** `x-goog-resource-id` in Watch API is the channel ID, not the file ID. The actual changed file ID requires calling the Drive Changes API. For simplicity, the Python script fetches the filename via Drive API using the file ID from a separate lookup. Update `extract_file.py` to handle this in the next step.
